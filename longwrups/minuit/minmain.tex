\def\part{\cleardoublepage
   \thispagestyle{plain}%
  \if@twocolumn
     \onecolumn
     \@tempswatrue
    \else \@tempswafalse
  \fi
  \hbox{}\vfil
  \secdef\@part\@spart}

\def\@part[#1]#2{\ifnum \c@secnumdepth >-2\relax
        \refstepcounter{part}%

        \addcontentsline{toc}{part}{\thepart
        \hspace{1em}#1}\else

        \addcontentsline{toc}{part}{#1}\fi
   \markboth{}{}%
   {\centering
    \interlinepenalty \@M
    \ifnum \c@secnumdepth >-2\relax
      \huge\bf \partname~\thepart
    \par
    \vskip 20\p@\fi
    \Huge \bf
    #2\par}\@endpart}

\def\@endpart{\vfil\newpage
   \if@twoside
       \hbox{}%
       \thispagestyle{empty}%
       \newpage
   \fi
   \if@tempswa
     \twocolumn
   \fi}
