
\DEFCMD{M}{ECHO}{MYMENU}{ECHO}{ string [ number minus positive option charset format ]}

\BEGARG
\DEFARG{STRING}{C}{Mandatory string}{}
\DEFARG{NUMBER}{I}{Small negative number}{ D=0 R=-10:0}
\DEFARG{MINUS}{C}{Optional string}{ D='MN' Minus}
\DEFARG{POSITIVE}{R}{Positive real}{ D=1.23 R=0:}
\DEFARG{OPTION}{C}{Implicit option}{ D=\EMPTY{} Minus}
\DEFARG{CHARSET}{C}{Self documenting option values in Range attribute}{ D='NATIVE'}
\DEFARG{FORMAT}{C}{Formatting rules}{ D='LINE'}
\ENDARG
\BEGOPT{OPTION}
\DEFOPT{\EMPTY}{\par
Blank option value
}
\DEFOPT{-}{\par
Option value '-' can make trouble.
}
\DEFOPT{1}{\par
Same is true for digits.
}
\ENDOPT
\BEGOPT{CHARSET}
\DEFOPT{NATIVE}{}
\DEFOPT{ASCII}{}
\DEFOPT{EBCDIC}{}
\ENDOPT
\BEGOPT{FORMAT}
\DEFOPT{LINE}{\par
The full explanation text has to be on a single logical line. The underscore 
can be used to join several physical lines into one logical line.
}
\DEFOPT{BLANKS}{\par
Leading blanks are ignored. Indentation can be used to make the CDF more 
readable. Multiple blanks are replaced by a single blank.
}
\ENDOPT

   \par
Echo command line arguments to demonstrate exception rules for option 
   assignments.  

\ENDCMD
