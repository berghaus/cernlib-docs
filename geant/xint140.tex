\Makehead{The General Control Commands}
\Ssbox{KINE  [ikine pkine]}
Set the variables in \Cind{/GCFLAG/}  {\tt IKINE, PKINE(10)}
 
Default : {\tt ikine=1 }
\Ssbox{TRACKING}
Restart tracking, clearing the tracking and hits
banks, but keeping the kinematics.
 
\Ssbox{TRIGGER }
Start a new event
\Ssbox{SWITCH  iswitch ival }
Change one element of array {\tt ISWIT(10)} in\Cind{ /GCFLAG/}
\Ssubr{DZSURV  bank-name [bank-number] 'See ZEBRA User Guide'}
{\tt CALL DZSURV('DZSURV'},ixstor,lzloc(ixstor,bank-name,bank-number))
\Ssbox{DZSHOW  bank-name [bank-number chopt] 'See ZEBRA User Guide'}
{\tt CALL DZSHOW('DZBANK',}ixstor,lzloc(ixstor,bank-name,bank-number),chopt ,
0,0,0,0)
 
Defaults are : {\tt bank-number=0  chopt='BHV'}
 
\Ssbox{DZSNAP  [idiv chopt] 'See ZEBRA User Guide'}
{\tt CALL DZSNAP('DZSNAP',idiv,chopt)}
 
Defaults are : {\tt idiv=2  chopt='M'}
 
\Ssbox{DZVERI  [idiv chopt] 'See ZEBRA User Guide'}
{\tt  CALL DZVERI('DZVERI',idiv,chopt)}
 
Defaults are : {\tt idiv=ixstor  chopt= 'CLSU'}
 
\Ssbox{DZAREA  [chopt ixstor] 'See ZEBRA User Guide'}
{\tt  CALL DZ('DZAREA',ixstor,chopt,lla,'N')}
 
Defaults are : {\tt ixstor=0  chopt=' '}
\Ssbox{DZSTOR  [ixstor] 'See ZEBRA User Guide'}
{\tt  CALL DZSTOR('DZSTOR',ixstor)}
 
Defaults is : {\tt ixstor=0}
\Ssbox{MZLOGL  level 'See ZEBRA User Guide'}
{\tt  CALL MZLOGL(0,level)}
 
Default is : {\tt level=0}
\Ssbox{FZLOGL  lun level 'See ZEBRA User Guide'}
{\tt  CALL FZLOGL(lun,level)}
 
Default is : {\tt lun=2  level=0}
\Ssbox{PRINT  name [number]}
{\tt CALL GPRINT(name,number)}
Default number = 0
\Ssbox{UNIT  lout }
to change lout in \Cind{/GCUNIT/}
 
Note: 5 and 11 are reserved units
\Ssbox{PHITS  [iuset iudet] }
to call \Rind{GPHITS(iuset,iudet)}
 
Defaults are : {\tt iuset=0  iudet=0}
\Ssbox{PDIGI  [iuset iudet]}
to call \Rind{GPDIGI(iuset,iudet)}
 
Defaults are : {\tt iuset=0  iudet=0}
\Ssbox{PRMAT  imate ipart mecan}
{\tt CALL GPRMAT(imate,ipart,mecan,90,elow)}
 
\Ssbox{PLMAT  imate ipart mecan [idm]}
{\tt  CALL GPLMAT(imate,ipart,mecan,90,elow,idm)}
 
Default is : {\tt idm=0}
\Ssbox{PSTAT}
{\tt  CALL GPSTAT}
\Ssbox{RETURN}
Return from the interactive routine \Rind{GINTER}.
 
Useful when called from the symbolic debugger.
\Ssbox{DEBUG  [ideb] }
{\tt If ideb='ON  ' then: idebug=1, idemin=1, idemax=1000000}\\
{\tt                else: idebug=0, idemin=0, idemax=0}
 
Default {\tt ideb='ON  '}
